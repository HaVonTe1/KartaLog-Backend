```kotlin
package io.github.havonte1.tcgwatcher.backend.config

import io.github.resilience4j.circuitbreaker.CircuitBreaker
import io.github.resilience4j.circuitbreaker.CircuitBreakerRegistry
import io.github.oshai.kotlinlogging.KotlinLogging
import org.junit.jupiter.api.Test
import org.junit.jupiter.api.extension.ExtendWith
import org.mockito.Mock
import org.mockito.junit.jupiter.MockitoExtension
import org.springframework.boot.actuate.health.Health
import org.springframework.boot.actuate.health.Status

@ExtendWith(MockitoExtension::class)
class CardMarketHealthIndicatorTest(
    @Mock private val circuitBreakerRegistry: CircuitBreakerRegistry,
    @Mock private val circuitBreaker: CircuitBreaker
) {
    private val logger = KotlinLogging.logger {}

    @Test
    fun `health returns UP when circuit breaker is CLOSED`() {
        whenever(circuitBreakerRegistry.circuitBreaker("cardMarketCircuitBreaker"))
            .thenReturn(circuitBreaker)
        whenever(circuitBreaker.state).thenReturn(CircuitBreaker.State.CLOSED)

        val indicator = CardMarketHealthIndicator(circuitBreakerRegistry)
        val health = indicator.health()

        assert(health.status == Status.UP)
        assert(health.details["circuitBreaker"] == "CLOSED")
    }

    @Test
    fun `health returns DOWN when circuit breaker is OPEN`() {
        whenever(circuitBreakerRegistry.circuitBreaker("cardMarketCircuitBreaker"))
            .thenReturn(circuitBreaker)
        whenever(circuitBreaker.state).thenReturn(CircuitBreaker.State.OPEN)

        val indicator = CardMarketHealthIndicator(circuitBreakerRegistry)
        val health = indicator.health()

        assert(health.status == Status.DOWN)
        assert(health.details["circuitBreaker"] == "OPEN")
    }

    @Test
    fun `health returns UNKNOWN when circuit breaker lookup fails`() {
        whenever(circuitBreakerRegistry.circuitBreaker("cardMarketCircuitBreaker"))
            .thenThrow(RuntimeException("Registry error"))

        val indicator = CardMarketHealthIndicator(circuitBreakerRegistry)
        val health = indicator.health()

        assert(health.status == Status.UNKNOWN)
    }
}
```
